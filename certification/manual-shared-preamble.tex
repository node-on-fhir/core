% Define maturity levels
\definecolor{draftcolor}{RGB}{255,165,0}
\definecolor{implementedcolor}{RGB}{100,149,237}
\definecolor{verifiedcolor}{RGB}{50,205,50}
\definecolor{validatedcolor}{RGB}{0,128,0}
\definecolor{gapcolor}{RGB}{205,50,50}

\newcommand{\maturitylabel}[1]{%
  \ifnum\pdfstrcmp{#1}{draft}=0
    \textcolor{draftcolor}{\textbf{DRAFT}}
  \else\ifnum\pdfstrcmp{#1}{implemented}=0
    \textcolor{implementedcolor}{\textbf{IMPLEMENTED}}
  \else\ifnum\pdfstrcmp{#1}{verified}=0
    \textcolor{verifiedcolor}{\textbf{VERIFIED (behavioral tests green)}}
  \else\ifnum\pdfstrcmp{#1}{validated}=0
    \textcolor{validatedcolor}{\textbf{VALIDATED}}
  \else\ifnum\pdfstrcmp{#1}{gap}=0
    \textcolor{gapcolor}{\textbf{GAP (documented, behavioral tests red)}}
  \else
    \textbf{#1}
  \fi\fi\fi\fi\fi
}

% Gherkin syntax highlighting for BDD

}

\lstset{
  basicstyle=\ttfamily\small,
  breaklines=true,
  columns=fullflexible,
  frame=single,
  backgroundcolor=\color{gray!10},
  % Render the § in BDD .feature listings correctly (T1 would otherwise show "ğ").
  % NOTE: under tectonic/XeTeX, literate mappings only work for chars <= U+00FF;
  % curly quotes / em-dashes are SILENTLY DROPPED from listings, so keep the
  % .feature files ASCII-normalized (straight quotes, -- for em-dash).
  literate={§}{{\S}}1,
}
